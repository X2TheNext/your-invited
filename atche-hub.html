1
Create Supabase Tables
Go to supabase.com → your ATCHÉ project → SQL Editor and run the following:
-- Model tracker table
create table if not exists "atcheAppData" (
  id uuid default gen_random_uuid() primary key,
  "userId" uuid references auth.users not null,
  models jsonb default '[]',
  notes  jsonb default '[]',
  events jsonb default '[]',
  "createdAt" timestamptz default now()
);
alter table "atcheAppData" enable row level security;
create policy "auth_read"   on "atcheAppData" for select using (auth.uid() is not null);
create policy "auth_update" on "atcheAppData" for update using (auth.uid() is not null);
create policy "owner_write" on "atcheAppData" for insert with check (auth.uid() = "userId");

-- Venue CRM team access
create policy "auth_read_venues"   on atche_venues for select using (auth.uid() is not null);
create policy "auth_update_venues" on atche_venues for update using (auth.uid() = user_id);
